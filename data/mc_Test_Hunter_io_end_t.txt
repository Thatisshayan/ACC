// _huntertest.js - Hunter.io E2E Test Script
const axios = require('axios');
require('dotenv').config();

const HUNTER_API_KEY = process.env.HUNTER_API_KEY;
const BASE_URL = 'https://api.hunter.io/v2';

// Simulate the hunter module functions
const hunter = {
  async checkHealth() {
    try {
      const response = await axios.get(`${BASE_URL}/account`, {
        params: { api_key: HUNTER_API_KEY }
      });
      return { success: true, data: response.data };
    } catch (error) {
      return { 
        success: false, 
        error: error.response?.data || error.message,
        status: error.response?.status
      };
    }
  },

  async domainSearch(domain, limit = 5) {
    try {
      const response = await axios.get(`${BASE_URL}/domain-search`, {
        params: { 
          domain, 
          limit,
          api_key: HUNTER_API_KEY 
        }
      });
      return { success: true, data: response.data };
    } catch (error) {
      return { 
        success: false, 
        error: error.response?.data || error.message 
      };
    }
  },

  async findEmail(domain, firstName, lastName) {
    try {
      const response = await axios.get(`${BASE_URL}/email-finder`, {
        params: { 
          domain, 
          first_name: firstName,
          last_name: lastName,
          api_key: HUNTER_API_KEY 
        }
      });
      return { success: true, data: response.data };
    } catch (error) {
      return { 
        success: false, 
        error: error.response?.data || error.message 
      };
    }
  },

  async verifyEmail(email) {
    try {
      const response = await axios.get(`${BASE_URL}/email-verifier`, {
        params: { 
          email,
          api_key: HUNTER_API_KEY 
        }
      });
      return { success: true, data: response.data };
    } catch (error) {
      return { 
        success: false, 
        error: error.response?.data || error.message 
      };
    }
  }
};

async function runTests() {
  console.log('=== Hunter.io E2E Test Suite ===\n');
  console.log(`API Key: ${HUNTER_API_KEY ? `${HUNTER_API_KEY.substring(0, 8)}...` : 'NOT SET'}`);
  console.log(`Key Length: ${HUNTER_API_KEY?.length || 0} chars\n`);

  // Test 1: checkHealth
  console.log('1. Testing checkHealth()...');
  const healthResult = await hunter.checkHealth();
  console.log(`   Result: ${healthResult.success ? '✅ PASS' : '❌ FAIL'}`);
  console.log(`   Response: ${JSON.stringify(healthResult.data || healthResult.error, null, 2)}`);
  
  if (!healthResult.success) {
    console.log('\n⚠️  checkHealth failed! Inspecting response...');
    if (healthResult.status === 401) {
      console.log('   → Authentication error. Check HUNTER_API_KEY validity.');
    } else if (healthResult.status === 404) {
      console.log('   → Endpoint not found. Verify API base URL.');
    } else {
      console.log('   → Unknown error. Check API connectivity.');
    }
  }

  // Test 2: domainSearch
  console.log('\n2. Testing domainSearch("stripe.com", 5)...');
  const domainResult = await hunter.domainSearch('stripe.com', 5);
  console.log(`   Result: ${domainResult.success ? '✅ PASS' : '❌ FAIL'}`);
  if (domainResult.success) {
    const emails = domainResult.data?.data?.emails || [];
    console.log(`   Emails found: ${emails.length}`);
    emails.forEach((email, i) => {
      console.log(`   ${i + 1}. ${email.value} (${email.type}) - ${email.confidence}% confidence`);
    });
  } else {
    console.log(`   Error: ${JSON.stringify(domainResult.error)}`);
  }

  // Test 3: findEmail
  console.log('\n3. Testing findEmail("stripe.com", "Patrick", "Collison")...');
  const emailResult = await hunter.findEmail('stripe.com', 'Patrick', 'Collison');
  console.log(`   Result: ${emailResult.success ? '✅ PASS' : '❌ FAIL'}`);
  if (emailResult.success) {
    const email = emailResult.data?.data;
    console.log(`   Email: ${email?.email || 'Not found'}`);
    console.log(`   Confidence: ${email?.confidence || 0}%`);
    console.log(`   Sources: ${email?.sources?.length || 0}`);
  } else {
    console.log(`   Error: ${JSON.stringify(emailResult.error)}`);
  }

  // Test 4: verifyEmail
  console.log('\n4. Testing verifyEmail("patrick@stripe.com")...');
  const verifyResult = await hunter.verifyEmail('patrick@stripe.com');
  console.log(`   Result: ${verifyResult.success ? '✅ PASS' : '❌ FAIL'}`);
  if (verifyResult.success) {
    const verification = verifyResult.data?.data;
    console.log(`   Status: ${verification?.status || 'Unknown'}`);
    console.log(`   Score: ${verification?.score || 0}`);
    console.log(`   Regex: ${verification?.regexp || false}`);
    console.log(`   Gibberish: ${verification?.gibberish || false}`);
    console.log(`   Disposable: ${verification?.disposable || false}`);
    console.log(`   Webmail: ${verification?.webmail || false}`);
    console.log(`   MX Records: ${verification?.mx_records || false}`);
    console.log(`   SMTP Server: ${verification?.smtp_server || false}`);
    console.log(`   SMTP Check: ${verification?.smtp_check || false}`);
  } else {
    console.log(`   Error: ${JSON.stringify(verifyResult.error)}`);
  }

  // Summary
  console.log('\n=== Test Summary ===');
  const tests = [
    { name: 'checkHealth', passed: healthResult.success },
    { name: 'domainSearch', passed: domainResult.success },
    { name: 'findEmail', passed: emailResult.success },
    { name: 'verifyEmail', passed: verifyResult.success }
  ];
  
  const passed = tests.filter(t => t.passed).length;
  const failed = tests.filter(t => !t.passed).length;
  
  tests.forEach(t => {
    console.log(`  ${t.passed ? '✅' : '❌'} ${t.name}`);
  });
  console.log(`\nTotal: ${passed}/4 passed, ${failed} failed`);
}

runTests().catch(console.error);
