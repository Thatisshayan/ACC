### 1. Add the `tryAlibaba` function (insert after `tryDeepSeek` function):

async function tryAlibaba(task) {
  try {
    const alibaba = require('../integrations/alibaba.js');
    if (alibaba.enabled()) {
      console.log('[provider] Trying Alibaba/Qwen...');
      const response = await alibaba.chat(task.instruction, 'qwen-plus');
      return {
        provider_used: 'alibaba',
        is_real_ai_result: true,
        cost_tier: 'low_cost',
        output: response.output || response,
        summary: response.summary || (typeof response === 'string' ? response.substring(0, 100) : 'Alibaba response')
      };
    }
  } catch (err) {
    console.warn('[provider] Alibaba/Qwen failed:', err.message);
  }
  return null;
}
