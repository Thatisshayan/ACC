## Analysis

The current `scheduler.js` daily briefing has critical issues:
1. **Fake repetitive content** - Same hardcoded messages sent every day
2. **No real data integration** - Not reading from actual task/note/job files
3. **No daily variation** - Same AI tip, same weather, same everything

## Recommended Fix

Replace the entire `sendBriefingForUser` function with a data-driven approach:

javascript
// cloud/telegram/features/scheduler.js

const fs = require('fs');
const path = require('path');

const DATA_TASKS_DIR = './data/tasks';
const DATA_STORAGE_DIR = './data/storage';
const DATA_JOBS_FILE = './data/job-applications.json';

// 7 unique AI tips - one for each day of week
const AI_TIPS = [
  "💡 **Monday Tip**: Break big tasks into 25-min Pomodoro chunks",
  "💡 **Tuesday Tip**: Use the 2-minute rule - if it takes <2min, do it now",
  "💡 **Wednesday Tip**: Review your week goals mid-week to stay on track",
  "💡 **Thursday Tip**: Batch similar tasks together for efficiency",
  "💡 **Friday Tip**: End your week by planning next Monday's priorities",
  "💡 **Saturday Tip**: Reflect on wins this week - celebrate progress",
  "💡 **Sunday Tip**: Set 3 top priorities for the upcoming week"
];

function getRealTaskCount() {
  try {
    if (!fs.existsSync(DATA_TASKS_DIR)) return 0;
    const files = fs.readdirSync(DATA_TASKS_DIR).filter(f => f.endsWith('.json'));
    return files.length;
  } catch (e) {
    return 0;
  }
}

function getRealNotesCount(userId) {
  try {
    const userDir = path.join(DATA_STORAGE_DIR, userId || 'default');
    if (!fs.existsSync(userDir)) return 0;
    const files = fs.readdirSync(userDir).filter(f => f.endsWith('.md') || f.endsWith('.txt'));
    return files.length;
  } catch (e) {
    return 0;
  }
}

function getRealJobCount() {
  try {
    if (!fs.existsSync(DATA_JOBS_FILE)) return 0;
    const data = JSON.parse(fs.readFileSync(DATA_JOBS_FILE, 'utf8'));
    return Array.isArray(data) ? data.length : 0;
  } catch (e) {
    return 0;
  }
}

function getUpcomingTasks() {
  try {
    if (!fs.existsSync(DATA_TASKS_DIR)) return [];
    const files = fs.readdirSync(DATA_TASKS_DIR).filter(f => f.endsWith('.json'));
    const pending = [];
    for (const file of files) {
      const content = JSON.parse(fs.readFileSync(path.join(DATA_TASKS_DIR, file), 'utf8'));
      if (content.status === 'pending') {
        pending.push(content.title || content.name || file.replace('.json', ''));
      }
    }
    return pending.slice(0, 3); // max 3 items
  } catch (e) {
    return [];
  }
}

function sendBriefingForUser(chatId, type) {
  const now = new Date();
  const dateStr = now.toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
  const dayOfWeek = now.getDay(); // 0=Sun, 1=Mon, ..., 6=Sat
  
  const taskCount = getRealTaskCount();
  const notesCount = getRealNotesCount(chatId);
  const jobCount = getRealJobCount();
  const upcoming = getUpcomingTasks();
  const tip = AI_TIPS[dayOfWeek];
  
  let message = `📅 **Daily Briefing** - ${dateStr}\n\n`;
  message += `📊 **Tasks**: ${taskCount} active\n`;
  message += `📝 **Notes**: ${notesCount} saved\n`;
  message += `💼 **Jobs**: ${jobCount} tracked\n`;
  message += `🌤 **Weather**: Check your weather app\n`;
  message += `${tip}\n`;
  
  if (upcoming.length > 0) {
    message += `\n⏰ **Upcoming**: ${upcoming.join(', ')}\n`;
  }
  
  // Send via Telegram API
  // ... (existing send logic)
  
  return message;
}

module.exports = { sendBriefingForUser };


## Key Changes

1. **Real data sources**: Reads from actual file system paths
2. **Daily uniqueness**: Day-of-week selects different AI tip
3. **No fake weather**: Just tells user to check their app
4. **Dynamic counts**: Tasks, notes, jobs all read from real files
5. **Upcoming tasks**: Filters for pending status from real task files
6. **Date always current**: Uses `new Date()` each call

## Risks

- File paths must exist or function handles gracefully (returns 0)
- JSON parsing errors caught with try/catch
- User ID may be undefined - defaults to 'default' storage folder
